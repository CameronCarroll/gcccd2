\begin{homeworkProblem}[State Government (Sacramento)]
  \problemAnswer{
    \paragraph{} All states must uphold a `republican form' of government, but the three-branch system is not required. (All states, nevertheless, implement the three-branch system.) All states except Nebraska implement a bicameral legislature. In this bicameral legislature, the smaller, upper house is the Senate while the larger, lower house is the Assembly, House of Representatives or House of Delegates. \\[0.2cm]
    \begin{tabularx}{450pt}{l| c c c}
      \textbf{Branch} & Executive & Legislative & Judicial \\
      \textbf{Represented By} & Governor \& Cabinet & Congress & Supreme Court Justices \\
      \textbf{Term} & 4 years, two possible terms & (See Below) & 12 years \\[0.5cm]
      \hline\\
    \end{tabularx}
    \begin{tabularx}{450pt}{l| c c}
      \textbf{House} & Senate & Assembly \\
      \textbf{Represented By} & Senators, 40 in California & Representatives, 80 in California \\
      \textbf{Term} & \multicolumn{2}{c}{12 years total in either upper or lower house}\\
    \end{tabularx}
  }
\end{homeworkProblem}
